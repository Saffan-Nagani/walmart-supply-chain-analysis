-- Query 1: Preview the data
SELECT *
FROM Walmart
LIMIT 10;

-- Query 2: Total number of records
SELECT COUNT(*) AS total_records
FROM Walmart;

-- Query 3: Date range of the dataset
SELECT 
  MIN(DATE) AS earliest_date,
  MAX(DATE) AS latest_date
FROM Walmart;

-- Query 4: Top 10 stores by total revenue
SELECT 
  STORE,
  ROUND(SUM(WEEKLY_SALES), 2) AS total_sales
FROM Walmart
GROUP BY STORE
ORDER BY total_sales DESC
LIMIT 10;

-- Query 5: Bottom 5 stores by total revenue
SELECT 
  STORE,
  ROUND(SUM(WEEKLY_SALES), 2) AS total_sales
FROM Walmart
GROUP BY STORE
ORDER BY total_sales ASC
LIMIT 5;

-- Query 6: Holiday vs normal week sales comparison
SELECT 
  HOLIDAY_FLAG,
  ROUND(AVG(WEEKLY_SALES), 2) AS avg_weekly_sales,
  COUNT(*) AS number_of_weeks
FROM Walmart
GROUP BY HOLIDAY_FLAG;

-- Query 7: Top 5 stores most impacted by holidays
SELECT 
  STORE,
  ROUND(AVG(CASE WHEN HOLIDAY_FLAG = 1 THEN WEEKLY_SALES END), 2) AS holiday_avg,
  ROUND(AVG(CASE WHEN HOLIDAY_FLAG = 0 THEN WEEKLY_SALES END), 2) AS normal_avg,
  ROUND(AVG(CASE WHEN HOLIDAY_FLAG = 1 THEN WEEKLY_SALES END) - 
        AVG(CASE WHEN HOLIDAY_FLAG = 0 THEN WEEKLY_SALES END), 2) AS holiday_impact
FROM Walmart
GROUP BY STORE
ORDER BY holiday_impact DESC
LIMIT 5;

-- Query 8: Stores with highest demand volatility
SELECT
  STORE,
  ROUND(AVG(WEEKLY_SALES), 2) AS avg_sales,
  ROUND(MAX(WEEKLY_SALES), 2) AS max_sales,
  ROUND(MIN(WEEKLY_SALES), 2) AS min_sales,
  ROUND(MAX(WEEKLY_SALES) - MIN(WEEKLY_SALES), 2) AS sales_range
FROM Walmart
GROUP BY STORE
ORDER BY sales_range DESC
LIMIT 5;

-- Query 9: Impact of unemployment on weekly sales
SELECT
  CASE
    WHEN UNEMPLOYMENT < 6 THEN 'Low Unemployment'
    WHEN UNEMPLOYMENT BETWEEN 6 AND 9 THEN 'Medium Unemployment'
    ELSE 'High Unemployment'
  END AS unemployment_bracket,
  ROUND(AVG(WEEKLY_SALES), 2) AS avg_weekly_sales,
  COUNT(*) AS number_of_records
FROM Walmart
GROUP BY unemployment_bracket
ORDER BY avg_weekly_sales DESC;
